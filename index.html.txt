    <script type="module">
        import { initializeApp } from 'https://www.gstatic.com/firebasejs/10.7.1/firebase-app.js';
        import { getAuth, createUserWithEmailAndPassword, signInWithEmailAndPassword, onAuthStateChanged, signOut } from 'https://www.gstatic.com/firebasejs/10.7.1/firebase-auth.js';
        import { getDatabase, ref, push, onValue, set, onDisconnect, serverTimestamp } from 'https://www.gstatic.com/firebasejs/10.7.1/firebase-database.js';

        // Firebase configuration
        const firebaseConfig = {
            apiKey: "AIzaSyD5ndtL_f3JZk6NTOMX1DlIGqO6roZWIkQ",
            authDomain: "buddy-chat-ced44.firebaseapp.com",
            databaseURL: "https://buddy-chat-ced44-default-rtdb.asia-southeast1.firebasedatabase.app",
            projectId: "buddy-chat-ced44",
            storageBucket: "buddy-chat-ced44.firebasestorage.app",
            messagingSenderId: "623200679884",
            appId: "1:623200679884:web:26c1c05ae0439f3709cf69",
            measurementId: "G-1TX2VDCD4F"
        };

        // Initialize Firebase
        const app = initializeApp(firebaseConfig);
        const auth = getAuth(app);
        const database = getDatabase(app);

        // Global variables
        let currentUser = null;
        let isSignUp = false;
        let typingTimeout = null;

        // Make functions globally accessible
        window.handleAuth = handleAuth;
        window.toggleAuthMode = toggleAuthMode;
        window.logout = logout;
        window.goBack = goBack;
        window.sendMessage = sendMessage;
        window.handleKeyPress = handleKeyPress;
        window.handleTyping = handleTyping;
        window.toggleDarkMode = toggleDarkMode;

        // Authentication functions
        function toggleAuthMode() {
            isSignUp = !isSignUp;
            const title = document.getElementById('authTitle');
            const toggleText = document.querySelector('.toggle-auth');
            
            if (isSignUp) {
                title.textContent = 'Create Account';
                toggleText.innerHTML = 'Already have an account? <a onclick="toggleAuthMode()">Sign In</a>';
            } else {
                title.textContent = 'Welcome to FireChat';
                toggleText.innerHTML = 'Don\'t have an account? <a onclick="toggleAuthMode()">Sign Up</a>';
            }
            document.getElementById('authError').textContent = '';
        }

        async function handleAuth() {
            const name = document.getElementById('loginName').value.trim();
            const email = document.getElementById('loginEmail').value.trim();
            const password = document.getElementById('loginPassword').value;
            const errorDiv = document.getElementById('authError');

            if (!name || !email || !password) {
                errorDiv.textContent = 'Please fill in all fields';
                return;
            }

            try {
                if (isSignUp) {
                    const userCredential = await createUserWithEmailAndPassword(auth, email, password);
                    await set(ref(database, 'users/' + userCredential.user.uid), {
                        name: name,
                        email: email,
                        online: true,
                        lastSeen: serverTimestamp()
                    });
                } else {
                    await signInWithEmailAndPassword(auth, email, password);
                }
                errorDiv.textContent = '';
            } catch (error) {
                errorDiv.textContent = error.message;
            }
        }

        function logout() {
            if (currentUser) {
                set(ref(database, 'users/' + currentUser.uid + '/online'), false);
            }
            signOut(auth);
        }

        function goBack() {
            if (currentUser) {
                set(ref(database, 'users/' + currentUser.uid + '/online'), false);
            }
            document.getElementById('authContainer').style.display = 'flex';
            document.getElementById('chatContainer').style.display = 'none';
            // Clear input fields
            document.getElementById('loginName').value = '';
            document.getElementById('loginEmail').value = '';
            document.getElementById('loginPassword').value = '';
            document.getElementById('authError').textContent = '';
        }

        // Auth state observer
        onAuthStateChanged(auth, async (user) => {
            if (user) {
                currentUser = user;
                
                // Get user data
                const userRef = ref(database, 'users/' + user.uid);
                onValue(userRef, (snapshot) => {
                    const userData = snapshot.val();
                    if (userData) {
                        setupChat(userData.name);
                        
                        // Set user online
                        set(ref(database, 'users/' + user.uid + '/online'), true);
                        onDisconnect(ref(database, 'users/' + user.uid + '/online')).set(false);
                    }
                }, { onlyOnce: true });
            } else {
                currentUser = null;
                document.getElementById('authContainer').style.display = 'flex';
                document.getElementById('chatContainer').style.display = 'none';
            }
        });

        function setupChat(userName) {
            document.getElementById('authContainer').style.display = 'none';
            document.getElementById('chatContainer').style.display = 'block';
            document.getElementById('userName').textContent = userName;
            document.getElementById('userAvatar').textContent = userName.charAt(0).toUpperCase();

            // Load messages
            loadMessages();
            
            // Load online users
            loadOnlineUsers();
            
            // Listen for typing indicators
            listenForTyping();
        }

        function loadMessages() {
            const messagesRef = ref(database, 'messages');
            onValue(messagesRef, (snapshot) => {
                const messagesContainer = document.getElementById('messagesContainer');
                messagesContainer.innerHTML = '';
                
                const messages = [];
                snapshot.forEach((childSnapshot) => {
                    messages.push({
                        id: childSnapshot.key,
                        ...childSnapshot.val()
                    });
                });

                messages.sort((a, b) => a.timestamp - b.timestamp);
                
                messages.forEach((message) => {
                    displayMessage(message);
                });

                scrollToBottom();
            });
        }

        function displayMessage(message) {
            const messagesContainer = document.getElementById('messagesContainer');
            const isOwn = message.userId === currentUser.uid;
            
            const messageDiv = document.createElement('div');
            messageDiv.className = `message ${isOwn ? 'own' : ''}`;
            
            const time = new Date(message.timestamp).toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });
            
            messageDiv.innerHTML = `
                <div class="message-avatar">${message.userName.charAt(0).toUpperCase()}</div>
                <div class="message-content">
                    <div class="message-header">
                        <span class="message-sender">${message.userName}</span>
                        <span class="message-time">${time}</span>
                    </div>
                    <div class="message-bubble">${escapeHtml(message.text)}</div>
                </div>
            `;
            
            messagesContainer.appendChild(messageDiv);
        }

        function sendMessage() {
            const input = document.getElementById('messageInput');
            const text = input.value.trim();
            
            if (!text || !currentUser) return;

            const userRef = ref(database, 'users/' + currentUser.uid);
            onValue(userRef, (snapshot) => {
                const userData = snapshot.val();
                
                push(ref(database, 'messages'), {
                    text: text,
                    userId: currentUser.uid,
                    userName: userData.name,
                    timestamp: Date.now()
                });
                
                input.value = '';
                stopTyping();
            }, { onlyOnce: true });
        }

        function handleKeyPress(event) {
            if (event.key === 'Enter') {
                sendMessage();
            }
        }

        function handleTyping() {
            if (!currentUser) return;
            
            set(ref(database, 'typing/' + currentUser.uid), {
                typing: true,
                timestamp: Date.now()
            });

            clearTimeout(typingTimeout);
            typingTimeout = setTimeout(stopTyping, 2000);
        }

        function stopTyping() {
            if (currentUser) {
                set(ref(database, 'typing/' + currentUser.uid), null);
            }
        }

        function listenForTyping() {
            const typingRef = ref(database, 'typing');
            onValue(typingRef, (snapshot) => {
                let someoneTyping = false;
                snapshot.forEach((childSnapshot) => {
                    const data = childSnapshot.val();
                    if (childSnapshot.key !== currentUser.uid && data.typing) {
                        someoneTyping = true;
                    }
                });
                
                document.getElementById('typingIndicator').style.display = someoneTyping ? 'block' : 'none';
            });
        }

        function loadOnlineUsers() {
            const usersRef = ref(database, 'users');
            onValue(usersRef, (snapshot) => {
                const onlineUsersDiv = document.getElementById('onlineUsers');
                onlineUsersDiv.innerHTML = '';
                let count = 0;
                
                snapshot.forEach((childSnapshot) => {
                    const user = childSnapshot.val();
                    if (user.online && childSnapshot.key !== currentUser.uid) {
                        count++;
                        const userDiv = document.createElement('div');
                        userDiv.className = 'online-user';
                        userDiv.innerHTML = `
                            <div class="status-dot"></div>
                            <div>${user.name}</div>
                        `;
                        onlineUsersDiv.appendChild(userDiv);
                    }
                });
                
                document.getElementById('onlineCount').textContent = count + 1;
            });
        }

        function scrollToBottom() {
            const container = document.getElementById('messagesContainer');
            container.scrollTop = container.scrollHeight;
        }

        function toggleDarkMode() {
            document.body.classList.toggle('dark-mode');
        }

        function escapeHtml(text) {
            const div = document.createElement('div');
            div.textContent = text;
            return div.innerHTML;
        }

        // Request notification permission
        if ('Notification' in window && Notification.permission === 'default') {
            Notification.requestPermission();
        }
    </script>
